
CC = gcc
CFLAGS = -Wall
TARGET = nav.+x
SOURCE = nav.c
INSTALL_DIR = /usr/+x

all: $(TARGET)

$(TARGET): $(SOURCE)
	$(CC) $(CFLAGS) -o $(TARGET) $(SOURCE)

install: $(TARGET)
	@mkdir -p $(INSTALL_DIR)
	@mv $(TARGET) $(INSTALL_DIR)/
	@chmod +x $(INSTALL_DIR)/$(TARGET)
	@echo "Installed $(TARGET) to $(INSTALL_DIR) ðŸŽ‰"
	@echo "Ensure $(INSTALL_DIR) is in your PATH (e.g., add 'export PATH=\$$PATH:$(INSTALL_DIR)' to ~/.bashrc)"

clean:
	@rm -f $(TARGET)
	@echo "Cleaned up build artifacts ðŸ§¹"

.PHONY: all install clean
